    <script>
        // Get the canvas element and its 2D rendering context
        const canvas = document.getElementById('characterCanvas');
        const ctx = canvas.getContext('2d');

        // Character state object to store selected options
        const characterState = {
            skin: 'white', // Default skin color
            hairType: 'straight', // Default hair type
            hairColor: 'black', // Default hair color
            eyeShape: 'round', // Default eye shape
            eyeColor: 'blue', // Default eye color
            outfit: 'hoodie' // Default outfit
        };

        // Base image URLs for different body parts.
        // !!! IMPORTANT: These are placeholder URLs. Replace them with paths to your actual image assets !!!
        // Example structure for your assets: assets/skin/skin-white.png, assets/hair/hair-straight-black.png, etc.
        const assetPaths = {
            skin: (color) => `https://placehold.co/400x400/${color === 'white' ? 'FFDAB9' : color === 'brown' ? 'A0522D' : color === 'black' ? '36454F' : color === 'chinese' ? 'F5DEB3' : 'E0BBE4'}/FFFFFF?text=Skin+${color}`,
            hair: (type, color) => `https://placehold.co/400x400/${color === 'black' ? '000000' : color === 'brown' ? '8B4513' : color === 'blonde' ? 'F5DEB3' : color === 'red' ? 'FF0000' : color === 'blue' ? '0000FF' : color === 'green' ? '008000' : color === 'purple' ? '800080' : 'FFC0CB'}/FFFFFF?text=Hair+${type}`,
            eyes: (shape, color) => `https://placehold.co/400x400/${color === 'brown' ? 'A0522D' : color === 'blue' ? '0000FF' : color === 'green' ? '008000' : color === 'black' ? '000000' : color === 'hazel' ? '8B7355' : color === 'grey' ? '808080' : '800080'}/FFFFFF?text=Eyes+${shape}`,
            outfit: (type) => `https://placehold.co/400x400/DDA0DD/000000?text=Outfit+${type}`,
            // You might need a base body image if skin color only changes the color of a base body.
            // For simplicity, this assumes skin images include the base body shape.
            // baseBody: 'assets/base/base-body.png' // A neutral base body if parts are layered on top
        };

        // Function to load an image and draw it on the canvas
        function loadImage(src) {
            return new Promise((resolve, reject) => {
                const img = new Image();
                img.onload = () => resolve(img);
                img.onerror = () => reject(new Error(`Failed to load image: ${src}`));
                img.src = src;
            });
        }

        // Function to draw the character on the canvas
        async function drawCharacter() {
            // Clear the canvas before drawing new elements
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // Set a background color for the character area
            ctx.fillStyle = '#f0f8ff'; // Light blue background
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            // Load and draw images in layers (bottom to top)
            // The order here is crucial for correct layering
            const layers = [
                // Base body (if separate from skin)
                // { type: 'baseBody', path: assetPaths.baseBody },

                // Skin (assumed to include the main body shape)
                { type: 'skin', path: assetPaths.skin(characterState.skin) },

                // Outfit (drawn over skin)
                { type: 'outfit', path: assetPaths.outfit(characterState.outfit) },

                // Hair (drawn over skin/body, potentially behind/around head)
                { type: 'hair', path: assetPaths.hair(characterState.hairType, characterState.hairColor) },

                // Eyes (drawn over skin/face)
                { type: 'eyes', path: assetPaths.eyes(characterState.eyeShape, characterState.eyeColor) },

                // Add more layers as needed (e.g., nose, mouth, accessories, eyebrows)
                // { type: 'nose', path: assetPaths.nose(characterState.noseShape) },
                // { type: 'mouth', path: assetPaths.mouth(characterState.mouthShape) },
            ];

            // Load all images concurrently
            const imagePromises = layers.map(layer => loadImage(layer.path));

            try {
                const loadedImages = await Promise.all(imagePromises);

                // Draw each image on the canvas
                loadedImages.forEach(img => {
                    // Draw image to fit canvas, maintaining aspect ratio
                    // You might need to adjust positions and sizes based on your actual assets
                    const aspectRatio = img.width / img.height;
                    let drawWidth = canvas.width;
                    let drawHeight = canvas.width / aspectRatio;

                    if (drawHeight > canvas.height) {
                        drawHeight = canvas.height;
                        drawWidth = canvas.height * aspectRatio;
                    }

                    const x = (canvas.width - drawWidth) / 2;
                    const y = (canvas.height - drawHeight) / 2;

                    ctx.drawImage(img, x, y, drawWidth, drawHeight);
                });
            } catch (error) {
                console.error("Error drawing character:", error);
                showMessage("Oops!", "Could not load some character parts. Please check the image paths!");
            }
        }

        // Function to update character state and redraw
        function updateCharacter(category, value) {
            characterState[category] = value;
            drawCharacter();
            updateSelectionUI(category, value); // Update UI to show selected option
        }

        // Function to update the selected state in the UI
        function updateSelectionUI(category, selectedValue) {
            // Remove 'selected' class from all options in the category
            document.querySelectorAll(`[data-category="${category}"]`).forEach(el => {
                el.classList.remove('selected');
            });
            // Add 'selected' class to the newly selected option
            const selectedElement = document.querySelector(`[data-category="${category}"][data-value="${selectedValue}"]`);
            if (selectedElement) {
                selectedElement.classList.add('selected');
            }
        }

        // Event Listeners for customization options
        document.getElementById('skinColorOptions').addEventListener('click', (event) => {
            const target = event.target.closest('.color-swatch');
            if (target) {
                updateCharacter('skin', target.dataset.value);
            }
        });

        document.getElementById('hairTypeOptions').addEventListener('click', (event) => {
            const target = event.target.closest('.option-button');
            if (target) {
                updateCharacter('hairType', target.dataset.value);
            }
        });

        document.getElementById('hairColorOptions').addEventListener('click', (event) => {
            const target = event.target.closest('.color-swatch');
            if (target) {
                updateCharacter('hairColor', target.dataset.value);
            }
        });

        document.getElementById('eyeShapeOptions').addEventListener('click', (event) => {
            const target = event.target.closest('.option-button');
            if (target) {
                updateCharacter('eyeShape', target.dataset.value);
            }
        });

        document.getElementById('eyeColorOptions').addEventListener('click', (event) => {
            const target = event.target.closest('.color-swatch');
            if (target) {
                updateCharacter('eyeColor', target.dataset.value);
            }
        });

        document.getElementById('outfitOptions').addEventListener('click', (event) => {
            const target = event.target.closest('.option-button');
            if (target) {
                updateCharacter('outfit', target.dataset.value);
            }
        });

        // Randomize Button
        document.getElementById('randomizeButton').addEventListener('click', () => {
            const skinColors = ['white', 'brown', 'black', 'chinese', 'vitiligo'];
            const hairTypes = ['straight', 'short', 'medium', 'long', 'afro', 'braids', 'hijab', 'curly', 'wavy'];
            const hairColors = ['black', 'brown', 'blonde', 'red', 'blue', 'green', 'purple', 'pink'];
            const eyeShapes = ['almond', 'round', 'hooded', 'monolid', 'upturned', 'downturned']; // Fixed ' hooded'
            const eyeColors = ['brown', 'blue', 'green', 'black', 'hazel', 'grey', 'purple'];
            const outfits = ['cultural', 'hoodie', 'dress'];

            characterState.skin = skinColors[Math.floor(Math.random() * skinColors.length)];
            characterState.hairType = hairTypes[Math.floor(Math.random() * hairTypes.length)];
            characterState.hairColor = hairColors[Math.floor(Math.random() * hairColors.length)];
            characterState.eyeShape = eyeShapes[Math.floor(Math.random() * eyeShapes.length)];
            characterState.eyeColor = eyeColors[Math.floor(Math.random() * eyeColors.length)];
            characterState.outfit = outfits[Math.floor(Math.random() * outfits.length)];

            drawCharacter();
            // Update UI for all categories after randomization
            updateSelectionUI('skin', characterState.skin);
            updateSelectionUI('hairType', characterState.hairType);
            updateSelectionUI('hairColor', characterState.hairColor);
            updateSelectionUI('eyeShape', characterState.eyeShape);
            updateSelectionUI('eyeColor', characterState.eyeColor);
            updateSelectionUI('outfit', characterState.outfit);
        });

        // Save Button (example functionality: download image)
        document.getElementById('saveButton').addEventListener('click', () => {
            const dataURL = canvas.toDataURL('image/png'); // Get image data as PNG
            const link = document.createElement('a');
            link.href = dataURL;
            link.download = 'my-awesome-character.png'; // Suggested filename
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
            showMessage("Character Saved!", "Your character has been downloaded as a PNG image!");
        });

        // --- Message Box Functions (to replace alert()) ---
        const messageBox = document.getElementById('messageBox');
        const messageBoxTitle = document.getElementById('messageBoxTitle');
        const messageBoxContent = document.getElementById('messageBoxContent');
        const messageBoxClose = document.getElementById('messageBoxClose');

        function showMessage(title, content) {
            messageBoxTitle.textContent = title;
            messageBoxContent.textContent = content;
            messageBox.style.display = 'flex'; // Show the message box
        }

        messageBoxClose.addEventListener('click', () => {
            messageBox.style.display = 'none'; // Hide the message box
        });
        // --- End Message Box Functions ---

        // Initial drawing of the character and UI update when the page loads
        document.addEventListener('DOMContentLoaded', () => {
            drawCharacter();
            // Set initial selections in UI
            updateSelectionUI('skin', characterState.skin);
            updateSelectionUI('hairType', characterState.hairType);
            updateSelectionUI('hairColor', characterState.hairColor);
            updateSelectionUI('eyeShape', characterState.eyeShape);
            updateSelectionUI('eyeColor', characterState.eyeColor);
            updateSelectionUI('outfit', characterState.outfit);
